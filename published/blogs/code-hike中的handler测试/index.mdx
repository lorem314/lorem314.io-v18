<Section title="callout 测试">

另类注释。

```js noLineNumber
const lorem = ipsum(dolor, sit)
// !callout[/amet/] This is a callout
const [amet, consectetur] = [0, 0]
lorem.adipiscing((sed, elit) => {
  if (sed) {
    amet += elit
  }
})
```

</Section>

<Section title="mark 测试">

高亮某一行代码，可自定义颜色。

```js
function lorem(ipsum, dolor = 1) {
  // !mark
  return dolor
}

function ipsum(lorem, dolor = 1) {
  // !mark(1:2) gold
  const sit = lorem == null ? 0 : lorem.sit
  dolor = sit - amet(dolor)
  // !mark[/sit/] pink
  return sit ? consectetur(lorem) : []
}
```

</Section>

<Section title="copy-button 测试">

默认情况下，代码块有复制按钮，通过添加 noCopyButton 来隐藏复制按钮。

```js noCopyButton
function lorem(ipsum, dolor = 1) {
  const sit = ipsum == null ? 0 : ipsum.sit
  dolor = sit - amet(dolor)
  return sit ? consectetur(ipsum) : []
}
```

</Section>

<Section title="link 测试">

可在代码中添加可点击的超链接。

```js
function lorem(ipsum, dolor = 1) {
  const sit = ipsum == null ? 0 : ipsum.sit
  dolor = sit - amet(dolor)
  // !link[/ipsum/] https://example.com

}
```

```js
// !link[/"(http.*?)"/gm]

  ]
  return links
```

</Section>

<Section title="class-name 测试">

在代码块中为行或列添加 className 从而添加样式。

```js
function lorem(ipsum, dolor = 1) {
  // !className line-through
  const sit = ipsum == null ? 0 : ipsum.sit
  dolor = sit - amet(dolor)
  // !className[/sit/] bg-red-700 rounded-lg px-1
  return sit ? consectetur(ipsum) : []
}
```
